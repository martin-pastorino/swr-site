<h1 style={{ textAlign: 'center' }}>Blog Posts</h1>

import Link from 'next/link'

<div style={{ margin: '20px 0', minHeight: 'calc(100vh - 350px)' }}>
  {
    [].concat(...NEXTRA_PAGE_MAP.map(item => {
      if (item.route === '/blog') {
        let meta, pages = {}
        for (const page of item.children) {
          if (page.name === 'meta.json') {
            meta = page.meta
          } else {
            if (page.name !== 'posts') pages[page.name] = page
          }
        }
        return Object.entries(meta).map(([name, info]) => {
          if (name === 'posts') return null
          const description = pages[name]?.frontMatter?.description || ''
          return <div key={name} style={{ marginBottom: 50 }}>
            <h3>
              <Link href={pages[name].route}>{typeof info === 'string' ? info : info.title}</Link>
            </h3>
            <p>{description}</p>
          </div>
        })
      }
      return null
    }))
  }
</div>
